# Start
Message -> TagPrefix SourcePrefix Command CrLf .

# Non-Terminals
AlphaNumeric -> Letter | Digit .
Integer -> Digit Integer .
Letter -> LowerCase | UpperCase .
LowerCase ->  LowerCaseHexLetter | LowerCaseLetter .
UpperCase ->  UpperCaseHexLetter | UpperCaseLetter .
LowerCaseLetter -> 0x67 | 0x68 | 0x69 | 0x6A | 0x6B | 0x6C | 0x6D | 0x6E | 0x6F | 0x70 | 0x71 | 0x72 | 0x73 | 0x74 | 0x75 | 0x76 | 0x77 | 0x78 | 0x79 | 0x7A .
UpperCaseLetter -> 0x47 | 0x48 | 0x49 | 0x4A | 0x4B | 0x4C | 0x4D | 0x4E | 0x4F | 0x50 | 0x51 | 0x52 | 0x53 | 0x54 | 0x55 | 0x56 | 0x57 | 0x58 | 0x59 | 0x5A .
LowerCaseHexLetter -> 0x61 | 0x62 | 0x63 | 0x64 0x65 | 0x66 .
UpperCaseHexLetter -> 0x41 | 0x42 | 0x43 | 0x44 | 0x45 | 0x46 .
Digit -> 0x30 | 0x31 | 0x32 | 0x33 | 0x34 | 0x35 | 0x36 | 0x37 | 0x38 | 0x39 .
NonPrintableWithoutNullLfCrBellFormatCTCP -> 0x05 | 0x06 | 0x08 | 0x09 | 0x0B | 0x0C | 0x0E | 0x10 | 0x12 | 0x13 | 0x14 | 0x15 | 0x16 | 0x17 | 0x18 | 0x19 | 0x1A | 0x1B | 0x1C | 0x7F .
UndefinedPunctuationMark -> 0x22 | 0x27 | 0x28 | 0x29 | 0x3C | 0x3E | 0x3F .
Special -> 0x5B | 0x5C | 0x5D | 0x5E | 0x5F | 0x60 | 0x7B | 0x7C | 0x7D .
CrLf -> 0x0D 0x0A | 0x0A .
Spaces -> 0x20 Spaces .
ShortName -> AlphaNumeric ShortNameSuffix .
ShortNameSuffix -> ShortNameList | .
ShortNameList -> AlphaNumeric ShortNameList | 0x2D ShortNameList .
ColorFormat -> 0x03 ColorCombination .
ColorCombination -> ForegroundColor ColorCombinationSuffix | .
ColorCombinationSuffix -> 0x2C BackgroundColor | .
Color -> Digit ColorSuffix .
ForegroundColor -> Color .
BackgroundColor -> Color .
ColorSuffix -> Digit | .
HexDigit -> LowerCaseHexLetter | UpperCaseHexLetter | Digit .
HexDecimal -> HexDigit HexDigit .
HexColorFormat -> 0x04 HexColorCombination .
HexColorCombination -> ForegroundHexColor HexColorCombinationSuffix | .
HexColorCombinationSuffix -> 0x2C BackgroundHexColor | .
HexColor -> HexDecimal HexDecimal HexDecimal .
ForegroundHexColor -> HexColor .
BackgroundHexColor -> HexColor .

CTCP -> 0x01 .
Bold -> 0x02 .
Reset -> 0x0F .
Monospace -> 0x11 .
Italics -> 0x1D .
Strikethrough -> 0x1E .
Underline -> 0x1F .
Format -> Bold | Italics | Underline | Strikethrough | Monospace | Reset | ColorFormat | HexColorFormat .

# Terminal collections
UTF8WithoutAlphaNumericFormatCTCPNullCrLFBase -> NonPrintableWithoutNullLfCrBellFormatCTCP | UndefinedPunctuationMark | 0x21 | 0x25 | 0x2B | 0x2D | 0x2E | 0x2F | 0x3D | Special | 0x7E | 0x2A | 0x23 | 0x26 | 0x24 | 0x80-0xFF .
UTF8WithoutCTCPNullCrLFBase -> UTF8WithoutAlphaNumericFormatCTCPNullCrLFBase | AlphaNumeric | 0x02 | 0x03 | 0x04 | 0x0F | 0x11 | 0x1D | 0x1E | 0x1F .
UTF8WithoutNullCrLFBase -> UTF8WithoutCTCPNullCrLFBase | 0x01 .
UTF8WithoutNullCrLfSemiColonSpace -> UTF8WithoutNullCrLFBase | 0x07 | 0x2C | 0x3A | 0x40 .
UTF8WithoutAlphaNumericFormatCTCPNullCrLfSpaceCommaColon -> UTF8WithoutAlphaNumericFormatCTCPNullCrLFBase | 0x07 | 0x3B | 0x40 .

# Command base
MiddlePrefixBaseList -> UTF8WithoutAlphaNumericFormatCTCPNullCrLfSpaceCommaColon MiddlePrefixBaseList | Format MiddlePrefixBaseList | LowerCaseLetter MiddlePrefixBaseList | UpperCaseLetter MiddlePrefixBaseList | 0x01 MiddlePrefixBaseList .
MiddlePrefixList -> MiddlePrefixBaseList MiddlePrefixList | Digit MiddlePrefixList | UpperCaseHexLetter MiddlePrefixList | LowerCaseHexLetter MiddlePrefixList | 0x2C MiddlePrefixList .
#TODO: implement
MiddlePrefixWithColonList -> MiddlePrefixList MiddlePrefixWithColonList | 0x3A MiddlePrefixWithColonList .
MiddleSuffix -> MiddlePrefixWithColonList | .
Middle -> MiddlePrefixList MiddleSuffix .
Trailing -> 0x3A TrailingPrefix .
TrailingPrefix -> TrailingList | .
TrailingList -> MiddlePrefixWithColonList TrailingList | 0x20 TrailingList .

# Error handling
Command -> ErroneousMessage | .
ErroneousMessage -> CommandName Params | CommandCode Params .
CommandName -> Letter CommandName .
CommandCode -> Digit Digit Digit .
Params -> ParamsPrefix | .
ParamsPrefix -> Spaces ParamsSuffix .
ParamsSuffix -> Middle ParamsPrefix | Trailing | .

# IRCv3 Tags
TagPrefix -> 0x40 Tags Spaces | .
Tags -> Tag TagsSuffix .
TagsSuffix -> TagsList | .
TagsList -> 0x3B Tag TagsList .
Tag -> TagKey TagSuffix .
TagSuffix -> 0x3D TagValue | .
TagKey -> 0x2B ShortName TagKeySuffix | ShortName TagKeySuffix .
TagKeySuffix ->  0x2F ShortName | .
TagValue -> TagValueList | .
TagValueList -> UTF8WithoutNullCrLfSemiColonSpace TagValueList .

# Target prefix
SourcePrefix -> 0x3A SourcePrefixList Spaces | .
SourcePrefixTarget -> AlphaNumeric SourcePrefixTargetSuffix | Special SourcePrefixTargetSuffix .
SourcePrefixTargetSuffix -> SourcePrefixList | .
SourcePrefixList -> AlphaNumeric SourcePrefixList | Special SourcePrefixList | 0x21 SourcePrefixList | 0x40 SourcePrefixList | 0x2D SourcePrefixList | 0x2E SourcePrefixList | 0x7E SourcePrefixList .
